# 5. Анализ результатов A/A/B эксперимента

# Создание сводной таблицы: группы vs события
pivot_table = data_clean.pivot_table(
    index='group',
    columns='event',
    values='user',
    aggfunc='nunique',
    fill_value=0
)
# Удаляем событие Tutorial из анализа, так как оно не входит в основную воронку
if 'Tutorial' in pivot_table.columns:
    pivot_table.drop(columns=['Tutorial'], inplace=True)

# Добавляем общее число пользователей в каждой группе
users_per_group = data_clean.groupby('group')['user'].nunique()
pivot_table['Total_Users'] = users_per_group

# Функция для проверки статистической значимости разницы долей (Z-тест)
def z_test_proportions(successes, trials, alpha):
    p1 = successes[0] / trials[0]
    p2 = successes[1] / trials[1]
    p_pooled = (successes[0] + successes[1]) / (trials[0] + trials[1])
    se = np.sqrt(p_pooled * (1 - p_pooled) * (1/trials[0] + 1/trials[1]))
    z_score = (p1 - p2) / se
    p_value = (1 - stats.norm.cdf(abs(z_score))) * 2

    print(f"p-value: {p_value:.4f}, alpha: {alpha}")
    if p_value < alpha:
        print("Результат статистически значим. Гипотеза о равенстве долей отвергается.")
    else:
        print("Разница не является статистически значимой.")
    print("-" * 50)
    return p_value

# Параметры для теста
alpha_initial = 0.1
num_tests = len(pivot_table.columns) - 1 # Исключаем колонку Total_Users

# Поправка Шидака для множественной проверки гипотез
alpha_corrected = 1 - (1 - alpha_initial) ** (1 / num_tests)
print(f"Изначальный уровень значимости (alpha): {alpha_initial}")
print(f"Количество проверок гипотез: {num_tests}")
print(f"Исправленный уровень значимости по Шидаку: {alpha_corrected:.4f}\n")

# Сравнение групп: A(246) vs B(248), A'(247) vs B(248), Объединенная A/A(246_247) vs B(248)
comparisons = [
    ("Группа 246 vs Группа 248", 246, 248),
    ("Группа 247 vs Группа 248", 247, 248),
]

for name, g1, g2 in comparisons:
    print(f"\n--- Сравнение {name} ---")
    trials = [pivot_table.loc[g1, 'Total_Users'], pivot_table.loc[g2, 'Total_Users']]
    
    for event in ['MainScreenAppear', 'OffersScreenAppear', 'CartScreenAppear', 'PaymentScreenSuccessful']:
        successes = [pivot_table.loc[g1, event], pivot_table.loc[g2, event]]
        print(f"\nСобытие: {event}")
        z_test_proportions(successes, trials, alpha_corrected)
